编程题1：指针交换函数（基础）
题目描述： 编写一个函数 swap，使用指针交换两个整数的值。在主函数中输入两个整数，调用该函数后输出交换后的结果。
输入格式： 两个整数 a 和 b
输出格式： 交换后的两个整数
输入样例：
10 20
输出样例：
20 10
参考代码：
#include <iostream>
using namespace std;

void swap(int* a, int* b) {
    int temp = *a;
    *a = *b;
    *b = temp;
}

int main() {
    int a, b;
    cin >> a >> b;
    swap(&a, &b);
    cout << a << " " << b << endl;
    return 0;
}

编程题2：数组逆置（中等）
题目描述： 编写一个函数 reverseArray，使用指针将一个整数数组逆置。在主函数中输入数组长度n和n个整数，调用该函数后输出逆置后的数组。
输入格式： 第一行：整数n（数组长度，1 ≤ n ≤ 100） 第二行：n个整数
输出格式： 逆置后的数组元素，用空格分隔
输入样例：
5
1 2 3 4 5
输出样例：
5 4 3 2 1
参考代码：
#include <iostream>
using namespace std;

void reverseArray(int* arr, int n) {
    int* left = arr;
    int* right = arr + n - 1;
    while (left < right) {
        int temp = *left;
        *left = *right;
        *right = temp;
        left++;
        right--;
    }
}

int main() {
    int n;
    cin >> n;
    int arr[100];
    for (int i = 0; i < n; i++) {
        cin >> arr[i];
    }
    reverseArray(arr, n);
    for (int i = 0; i < n; i++) {
        cout << arr[i];
        if (i < n - 1) cout << " ";
    }
    cout << endl;
    return 0;
}

编程题3：指针查找最大值与最小值（综合）
题目描述： 编写一个函数 findMinMax，使用指针在数组中查找最大值和最小值，并通过指针参数返回它们的值和位置（下标）。
函数原型：
void findMinMax(int* arr, int n, int* maxVal, int* minVal, int* maxPos, int* minPos);
输入格式： 第一行：整数n（数组长度，1 ≤ n ≤ 100） 第二行：n个整数
输出格式： 四行，分别输出：
- 最大值
- 最大值的位置（下标）
- 最小值
- 最小值的位置（下标）
输入样例：
6
3 7 2 9 5 1
输出样例：
9
3
1
5
参考代码：
#include <iostream>
using namespace std;

void findMinMax(int* arr, int n, int* maxVal, int* minVal, int* maxPos, int* minPos) {
    *maxVal = *minVal = arr[0];
    *maxPos = *minPos = 0;
    
    for (int i = 1; i < n; i++) {
        if (*(arr + i) > *maxVal) {
            *maxVal = *(arr + i);
            *maxPos = i;
        }
        if (*(arr + i) < *minVal) {
            *minVal = *(arr + i);
            *minPos = i;
        }
    }
}

int main() {
    int n;
    cin >> n;
    int arr[100];
    for (int i = 0; i < n; i++) {
        cin >> arr[i];
    }
    
    int maxVal, minVal, maxPos, minPos;
    findMinMax(arr, n, &maxVal, &minVal, &maxPos, &minPos);
    
    cout << maxVal << endl;
    cout << maxPos << endl;
    cout << minVal << endl;
    cout << minPos << endl;
    
    return 0;
}
